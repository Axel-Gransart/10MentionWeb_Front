Les string en JavaScript

Modification d'un type -> string en type -> number

let chiffre = "12.5";
  console.log(typeof chiffre);
  console.log(chiffre);
      
  let chiffreDecimal = parseFloat(chiffre);
  console.log(typeof chiffreDecimal);
  console.log(chiffreDecimal);
      
  let chiffreEntier = parseInt(chiffre);
  console.log(typeof chiffreEntier);
  console.log(chiffreEntier);
